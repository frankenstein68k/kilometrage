CREATE TABLE MILEAGE_USER
(
USERNAME VARCHAR (20) PRIMARY KEY,
FNAME VARCHAR (50),
LNAME VARCHAR (50),
OUTPUT_PREF INT
)

CREATE TABLE CAR
(
USERNAME VARCHAR (40), 
CAR_MAKE VARCHAR (40),
CAR_MODEL VARCHAR (40),
CAR_YEAR INT,
CAR_ID INT IDENTITY(1,1) PRIMARY KEY,
FOREIGN KEY (USERNAME) REFERENCES MILEAGE_USER (USERNAME)
)

CREATE TABLE FILL_UP
(
CAR_ID INT,
FILL_UP_ID INT IDENTITY(1,1),
LITRES_IN INT,
GAS_STATION_BRAND VARCHAR (40)
FOREIGN KEY (CAR_ID) REFERENCES CAR (CAR_ID)
)
